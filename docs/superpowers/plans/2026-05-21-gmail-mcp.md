**Goal:** Local MCP server that gives Claude Code parallel access to two Gmail accounts (`company`, `personal`) for triage — search, read, draft, label/archive — sourced from refresh tokens stored on disk.

**Architecture:** Bun + TypeScript MCP server at `~/.claude/channels/gmail/`. One Google OAuth Desktop client shared across both accounts, each with its own refresh token in `tokens/<alias>.json`. Server exposes 5 tools (`gmail_search`, `gmail_get`, `gmail_draft`, `gmail_modify`, `gmail_labels`), each gated by an `account` argument.

**Tech Stack:** Bun, TypeScript, `@modelcontextprotocol/sdk`, `googleapis`.

```ts
export type ToolError = { error: string; code: string; hint?: string };

export function mapError(e: any, account: string): ToolError {
  const data = e?.response?.data;
  const gErr = data?.error;
  if (gErr === "invalid_grant" || gErr === "unauthorized_client") {
    return {
      error: `Authentication failed for account '${account}'`,
      code: "auth_failed",
      hint: `Run: cd ~/.claude/channels/gmail && bun run auth ${account}`,
    };
  }
  if (e?.code === 404) return { error: e.message ?? "Not found", code: "not_found" };
  if (e?.code === 429) return { error: "Rate limited", code: "rate_limited", hint: "Retry shortly" };
  if (e?.code === 400) return { error: e.message ?? "Bad request", code: "bad_request" };
  return { error: `Gmail upstream error: ${e?.message ?? String(e)}`, code: "upstream" };
}

export function badRequest(msg: string): ToolError {
  return { error: msg, code: "bad_request" };
}
```

```ts
export type DraftInput = {
  to: string;
  cc?: string;
  bcc?: string;
  subject: string;
  body: string;
  inReplyTo?: string;
  references?: string[];
};

function encodeHeader(value: string): string {
  // RFC 2047 encoded-word if non-ASCII present
  if (/^[\x00-\x7F]*$/.test(value)) return value;
  return `=?UTF-8?B?${Buffer.from(value, "utf8").toString("base64")}?=`;
}

export function buildRawMessage(d: DraftInput): string {
  const lines: string[] = [];
  lines.push(`To: ${d.to}`);
  if (d.cc) lines.push(`Cc: ${d.cc}`);
  if (d.bcc) lines.push(`Bcc: ${d.bcc}`);
  lines.push(`Subject: ${encodeHeader(d.subject)}`);
  if (d.inReplyTo) lines.push(`In-Reply-To: ${d.inReplyTo}`);
  if (d.references && d.references.length) lines.push(`References: ${d.references.join(" ")}`);
  lines.push(`MIME-Version: 1.0`);
  lines.push(`Content-Type: text/plain; charset="UTF-8"`);
  lines.push(`Content-Transfer-Encoding: 8bit`);
  const message = lines.join("\r\n") + "\r\n\r\n" + d.body;
  return Buffer.from(message, "utf8").toString("base64url");
}
```

```ts
import { promises as fs } from "node:fs";
import { homedir } from "node:os";
import { join } from "node:path";
import { google, gmail_v1 } from "googleapis";
import { OAuth2Client } from "google-auth-library";

export const CHANNEL_DIR = join(homedir(), ".claude/channels/gmail");
export const TOKENS_DIR = join(CHANNEL_DIR, "tokens");
export const CLIENT_PATH = join(CHANNEL_DIR, "oauth-client.json");

export const SCOPES = [
  "https://www.googleapis.com/auth/gmail.readonly",
  "https://www.googleapis.com/auth/gmail.modify",
  "https://www.googleapis.com/auth/gmail.compose",
];

export type Alias = "company" | "personal";

async function loadClient(): Promise<{ client_id: string; client_secret: string; redirect_uris: string[] }> {
  const raw = await fs.readFile(CLIENT_PATH, "utf8").catch(() => {
    throw new Error(`Missing OAuth client at ${CLIENT_PATH}. See spec section 'Setup flow'.`);
  });
  const j = JSON.parse(raw);
  const inner = j.installed ?? j.web;
  if (!inner) throw new Error("oauth-client.json must contain 'installed' or 'web' key");
  return { client_id: inner.client_id, client_secret: inner.client_secret, redirect_uris: inner.redirect_uris ?? [] };
}

export async function makeOAuth2(redirectUri?: string): Promise<OAuth2Client> {
  const c = await loadClient();
  return new google.auth.OAuth2(c.client_id, c.client_secret, redirectUri ?? c.redirect_uris[0] ?? "http://localhost");
}

export function tokenPath(alias: Alias): string {
  return join(TOKENS_DIR, `${alias}.json`);
}

export async function saveToken(alias: Alias, token: { refresh_token?: string | null }): Promise<void> {
  if (!token.refresh_token) throw new Error("No refresh_token returned. Re-run with prompt=consent.");
  await fs.mkdir(TOKENS_DIR, { recursive: true });
  await fs.writeFile(tokenPath(alias), JSON.stringify({ refresh_token: token.refresh_token }, null, 2), { mode: 0o600 });
}

export async function loadAuthClient(alias: Alias): Promise<OAuth2Client> {
  if (alias !== "company" && alias !== "personal") {
    throw Object.assign(new Error(`Unknown account alias: ${alias}`), { code: 400 });
  }
  const raw = await fs.readFile(tokenPath(alias), "utf8").catch(() => {
    throw Object.assign(new Error(`No token for '${alias}'. Run: bun run auth ${alias}`), {
      response: { data: { error: "invalid_grant" } },
    });
  });
  const { refresh_token } = JSON.parse(raw);
  const oauth2 = await makeOAuth2();
  oauth2.setCredentials({ refresh_token });
  return oauth2;
}

export async function gmailFor(alias: Alias): Promise<gmail_v1.Gmail> {
  const auth = await loadAuthClient(alias);
  return google.gmail({ version: "v1", auth });
}
```

### Task 5: Auth CLI (`bun run auth <alias>`)

**Files:**
- Create: `~/.claude/channels/gmail/auth.ts`

Loopback flow: spin a one-shot HTTP server on `127.0.0.1:<random>`, build a consent URL with that redirect, open the user's browser, capture the `code`, exchange for a refresh token, save, exit.

- [ ] **Step 1: Implement `auth.ts`**

```ts
#!/usr/bin/env bun
import { createServer } from "node:http";
import { AddressInfo } from "node:net";
import { makeOAuth2, saveToken, SCOPES, Alias } from "./gmail";

const alias = process.argv[2] as Alias;
if (alias !== "company" && alias !== "personal") {
  console.error("Usage: bun run auth <company|personal>");
  process.exit(2);
}

const server = createServer();
await new Promise<void>((r) => server.listen(0, "127.0.0.1", r));
const port = (server.address() as AddressInfo).port;
const redirectUri = `http://127.0.0.1:${port}/callback`;

const oauth2 = await makeOAuth2(redirectUri);
const url = oauth2.generateAuthUrl({
  access_type: "offline",
  prompt: "consent",
  scope: SCOPES,
  include_granted_scopes: true,
});

console.log(`\nAuthorize '${alias}' Gmail account:\n  ${url}\n`);
// Best-effort open in browser
const open = (u: string) => Bun.spawn(["open", u]).exited.catch(() => {});
open(url);

const code: string = await new Promise((resolve, reject) => {
  server.on("request", (req, res) => {
    const u = new URL(req.url ?? "/", `http://127.0.0.1:${port}`);
    if (u.pathname !== "/callback") {
      res.writeHead(404).end();
      return;
    }
    const c = u.searchParams.get("code");
    const err = u.searchParams.get("error");
    res.writeHead(200, { "Content-Type": "text/html" });
    res.end(`<html><body><h2>${c ? "Authorized — you can close this tab." : `Error: ${err}`}</h2></body></html>`);
    if (c) resolve(c);
    else reject(new Error(err ?? "no code"));
  });
});

server.close();
const { tokens } = await oauth2.getToken(code);
await saveToken(alias, tokens);
console.log(`Saved refresh token for '${alias}' → tokens/${alias}.json`);
process.exit(0);
```

```ts
import { buildRawMessage, DraftInput } from "./mime";

export async function searchMessages(alias: Alias, query: string, maxResults = 25, pageToken?: string) {
  const g = await gmailFor(alias);
  const list = await g.users.messages.list({
    userId: "me",
    q: query,
    maxResults,
    pageToken,
  });
  const ids = list.data.messages ?? [];
  if (ids.length === 0) return { messages: [], nextPageToken: list.data.nextPageToken ?? null };
  // Pull snippets in parallel
  const details = await Promise.all(
    ids.map((m) =>
      g.users.messages.get({ userId: "me", id: m.id!, format: "metadata", metadataHeaders: ["From", "Subject", "Date"] })
    )
  );
  return {
    messages: details.map((r) => {
      const headers = r.data.payload?.headers ?? [];
      const h = (n: string) => headers.find((x) => x.name?.toLowerCase() === n.toLowerCase())?.value ?? null;
      return {
        id: r.data.id,
        threadId: r.data.threadId,
        snippet: r.data.snippet,
        from: h("From"),
        subject: h("Subject"),
        date: h("Date"),
        labelIds: r.data.labelIds ?? [],
      };
    }),
    nextPageToken: list.data.nextPageToken ?? null,
  };
}

function decodeBody(payload: any): { text: string; html: string } {
  let text = "", html = "";
  const walk = (p: any) => {
    if (!p) return;
    const data = p.body?.data;
    if (data) {
      const decoded = Buffer.from(data, "base64url").toString("utf8");
      if (p.mimeType === "text/plain") text += decoded;
      else if (p.mimeType === "text/html") html += decoded;
    }
    (p.parts ?? []).forEach(walk);
  };
  walk(payload);
  return { text, html };
}

export async function getMessage(alias: Alias, messageId: string, format: "full" | "metadata" = "full") {
  const g = await gmailFor(alias);
  const r = await g.users.messages.get({ userId: "me", id: messageId, format });
  const headers = r.data.payload?.headers ?? [];
  const body = format === "full" ? decodeBody(r.data.payload) : { text: "", html: "" };
  const attachments = [] as { filename: string; mimeType: string; size: number; attachmentId: string }[];
  const collectAtt = (p: any) => {
    if (!p) return;
    if (p.filename && p.body?.attachmentId) {
      attachments.push({ filename: p.filename, mimeType: p.mimeType, size: p.body.size ?? 0, attachmentId: p.body.attachmentId });
    }
    (p.parts ?? []).forEach(collectAtt);
  };
  collectAtt(r.data.payload);
  return {
    id: r.data.id,
    threadId: r.data.threadId,
    labelIds: r.data.labelIds ?? [],
    snippet: r.data.snippet,
    headers: Object.fromEntries(headers.map((h: any) => [h.name, h.value])),
    body,
    attachments,
  };
}

export async function createDraft(alias: Alias, input: {
  to: string; cc?: string; bcc?: string; subject: string; body: string;
  replyToMessageId?: string; threadId?: string;
}) {
  const g = await gmailFor(alias);
  let inReplyTo: string | undefined;
  let references: string[] | undefined;
  let threadId = input.threadId;
  if (input.replyToMessageId) {
    const src = await g.users.messages.get({
      userId: "me", id: input.replyToMessageId, format: "metadata",
      metadataHeaders: ["Message-ID", "References"],
    });
    threadId = threadId ?? src.data.threadId ?? undefined;
    const headers = src.data.payload?.headers ?? [];
    const mid = headers.find((h: any) => h.name?.toLowerCase() === "message-id")?.value;
    const refs = headers.find((h: any) => h.name?.toLowerCase() === "references")?.value;
    if (mid) {
      inReplyTo = mid;
      references = refs ? refs.split(/\s+/).concat(mid) : [mid];
    }
  }
  const raw = buildRawMessage({
    to: input.to, cc: input.cc, bcc: input.bcc,
    subject: input.subject, body: input.body,
    inReplyTo, references,
  });
  const r = await g.users.drafts.create({
    userId: "me",
    requestBody: { message: { raw, threadId } },
  });
  return { draftId: r.data.id, messageId: r.data.message?.id, threadId: r.data.message?.threadId };
}

export async function modifyMessage(alias: Alias, messageId: string, addLabels?: string[], removeLabels?: string[]) {
  const g = await gmailFor(alias);
  // Resolve label names → IDs if names supplied (anything not matching /^Label_\d+|SYSTEM_LABEL/i style)
  const all = await listLabels(alias);
  const resolve = (xs?: string[]) =>
    (xs ?? []).map((x) => {
      const byName = all.find((l) => l.name?.toLowerCase() === x.toLowerCase());
      return byName?.id ?? x;
    });
  const r = await g.users.messages.modify({
    userId: "me",
    id: messageId,
    requestBody: { addLabelIds: resolve(addLabels), removeLabelIds: resolve(removeLabels) },
  });
  return { id: r.data.id, labelIds: r.data.labelIds ?? [] };
}

export async function listLabels(alias: Alias) {
  const g = await gmailFor(alias);
  const r = await g.users.labels.list({ userId: "me" });
  return (r.data.labels ?? []).map((l) => ({ id: l.id!, name: l.name!, type: l.type! }));
}
```

```ts
server.setRequestHandler(ListToolsRequestSchema, async () => ({ tools: TOOLS }));

server.setRequestHandler(CallToolRequestSchema, async (req) => {
  const { name, arguments: args = {} } = req.params;
  const a = (args as any).account as Alias;
  try {
    if (a !== "company" && a !== "personal") {
      return { content: [{ type: "text", text: JSON.stringify(badRequest("account must be 'company' or 'personal'")) }], isError: true };
    }
    let result: any;
    switch (name) {
      case "gmail_search":
        result = await searchMessages(a, (args as any).query, (args as any).maxResults, (args as any).pageToken);
        break;
      case "gmail_get":
        result = await getMessage(a, (args as any).messageId, (args as any).format);
        break;
      case "gmail_draft":
        result = await createDraft(a, args as any);
        break;
      case "gmail_modify":
        result = await modifyMessage(a, (args as any).messageId, (args as any).addLabels, (args as any).removeLabels);
        break;
      case "gmail_labels":
        result = await listLabels(a);
        break;
      default:
        return { content: [{ type: "text", text: JSON.stringify(badRequest(`Unknown tool: ${name}`)) }], isError: true };
    }
    return { content: [{ type: "text", text: JSON.stringify(result, null, 2) }] };
  } catch (e) {
    const err = mapError(e, a);
    return { content: [{ type: "text", text: JSON.stringify(err) }], isError: true };
  }
});

const transport = new StdioServerTransport();
await server.connect(transport);
console.error("gmail mcp: ready");
```
